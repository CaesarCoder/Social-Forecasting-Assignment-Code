detect_date_format <- function(date_string) {
  if (grepl("/", date_string)) {
    parts <- unlist(strsplit(date_string, "/"))
    if (length(parts) == 3) {
      if (nchar(parts[1]) == 4) {
        return("%Y/%m/%d")
      } else if (nchar(parts[3]) == 4) {
        return("%m/%d/%Y")
      }
    }
  } else if (grepl("-", date_string)) {
    return("%Y-%m-%d")
  }
  return(NULL)
}
